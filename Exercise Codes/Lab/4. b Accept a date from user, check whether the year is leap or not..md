# Accept a date from user, check whether the year is leap or not.
=================================================================
using System;
using System.Windows.Forms;

private void btnCheckLeap_Click(object sender, EventArgs e)
{
    int year = dtpDate.Value.Year;

    if (DateTime.IsLeapYear(year))
    {
        lblLeapResult.Text = year + " is a Leap Year";
    }
    else
    {
        lblLeapResult.Text = year + " is NOT a Leap Year";p
    }
}
