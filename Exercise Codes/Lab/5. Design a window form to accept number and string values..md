# Design a window form to accept number and string values from users, display the reverse of number and string using string 
# inbuilt functions.

using System;
using System.Linq;
using System.Windows.Forms;

namespace ReverseApp
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        private void btnReverse_Click(object sender, EventArgs e)
        {
            // Reverse Number
            string num = txtNumber.Text;
            string revNum = new string(num.Reverse().ToArray());

            // Reverse String
            string str = txtString.Text;
            string revStr = new string(str.Reverse().ToArray());

            // Display Results
            lblRevNumber.Text = "Reversed Number: " + revNum;
            lblRevString.Text = "Reversed String: " + revStr;
        }
    }
}
