# 2. Design a window form to accept some text and display the number of vowels, consonant and special symbols.  
==============================================================================================================
using System;
using System.Linq;
using System.Windows.Forms;

namespace TextAnalyzer
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        private void btnCount_Click(object sender, EventArgs e)
        {
            string text = txtInput.Text.ToLower();

            int vowels = 0;
            int consonants = 0;
            int special = 0;

            foreach (char ch in text)
            {
                if (char.IsLetter(ch))
                {
                    if ("aeiou".Contains(ch))
                        vowels++;
                    else
                        consonants++;
                }
                else if (!char.IsWhiteSpace(ch))
                {
                    special++;
                }
            }

            lblVowels.Text = "Vowels: " + vowels;
            lblConsonants.Text = "Consonants: " + consonants;
            lblSpecial.Text = "Special Symbols: " + special;
        }
    }
}
